228std::unordered_map<std::string, int> HelperUtils::decryptFiles(const std::vector<fs::path> &filePaths) {
229 CryptoDLL cryptoDll;
230 RestApiDLL restApiDll;
231 FileMarkDLL fileMarkDll;
232 std::unordered_map<std::string, int> results;
233 std::vector<FileData> fileDataVec;
234
235 for (const auto &filePath : filePaths) {
236 // Create the FileData structure
237 FileData fileData;
238 fileData.EncryptionID->resize(64);
239 fileData.FileID->resize(64);
240
241 fileMarkDll.extractIDsFromFile(filePath.wstring().c_str(), fileData.FileID->data(), fileData.EncryptionID->data());
242
243 if (!restApiDll.SearchEntry(fileData)) {
244 results.insert({"Search", 1}); // Error finding the entry
245 continue;
246 }
247 fileData.EncryptedFilePath->assign(filePath.wstring().begin(), filePath.wstring().end());
248 fileDataVec.push_back(fileData);
249 }
250
251 // Decrypt the files
252 std::vector<bool> decryptResults(fileDataVec.size(), false);
253 cryptoDll.DecryptFiles(fileDataVec.data(), static_cast<int>(fileDataVec.size()), decryptResults);
254
255 for (size_t i = 0; i < fileDataVec.size(); ++i) {
256 std::string algorithm(fileDataVec[i].AlgorithmenType->begin(), fileDataVec[i].AlgorithmenType->end());
257
258 // Check if decryption was successful
259 if (!decryptResults[i]) {
260 results.insert({algorithm, 2}); // Error during decryption
261 continue;
262 }
263
264 // Delete the entry from the database
265 if (!restApiDll.DeleteEntry(fileDataVec[i])) {
266 results.insert({algorithm, 3}); // Error deleting the entry
267 continue;
268 }
269
270 // Clean up the FileData structure
271 fileDataVec[i].Key->clear();
272 fileDataVec[i].Iv->clear();
273 results.insert({algorithm, -1}); // Success
274 }
275
276 return results;
277}

286std::vector<int> HelperUtils::encryptFiles(const std::vector<fs::path> &filePaths, const std::vector<std::string> &algorithm) {
287 CryptoDLL cryptoDll;
288 RestApiDLL restApiDll;
289 std::vector<int> results;
290 std::vector<FileData> fileDataVec;
291
292 for (int i = 0; i < filePaths.size(); i++) {
293 // Create the FileData structure
294 FileData fileData;
295 if (algorithm[i] == "AES-128") {
296 fileData = StructUtils::createFileDataStruct(globalDefinitions::AlgorithmType::AES128, filePaths[i]);
297 } else {
298 fileData = StructUtils::createFileDataStruct(globalDefinitions::AlgorithmType::AES256, filePaths[i]);
299 }
300 fileDataVec.push_back(fileData);
301 }
302
303 // Encrypt the file
304 std::vector<bool> encryptResults(fileDataVec.size(), false);
305 cryptoDll.EncryptFiles(fileDataVec.data(), static_cast<int>(fileDataVec.size()), encryptResults);
306
307 for (auto fileData : fileDataVec) {
308 // Check if encryption was successful
309 if (fileData.EncryptedFilePath->empty() || !encryptResults[0]) {
310 results.push_back(1); // Error during encryption
311 continue;
312 }
313
314 // Insert the entry into the database
315 if (!restApiDll.InsertEntry(fileData)) {
316 results.push_back(2); // Error inserting the entry
317 continue;
318 }
319
320 // Clean up the FileData structure
321 fileData.Key->clear();
322 fileData.Iv->clear();
323 results.push_back(-1); // Success
324 }
325
326 return results;
327}
