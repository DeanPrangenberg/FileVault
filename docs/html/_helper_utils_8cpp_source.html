226std::unordered_map<std::string, int> HelperUtils::decryptFiles(const std::vector<fs::path> &filePaths) {
227 CryptoDLL cryptoDll;
228 RestApiDLL restApiDll;
229 FileMarkDLL fileMarkDll;
230 std::unordered_map<std::string, int> results;
231 std::vector<FileData> fileDataVec;
232
233 for (const auto &filePath : filePaths) {
234 // Create the FileData structure
235 FileData fileData;
236 fileData.EncryptionID->resize(64);
237 fileData.FileID->resize(64);
238
239 fileMarkDll.extractIDsFromFile(&filePath, fileData.FileID.get(), fileData.EncryptionID.get());
240
241 if (!restApiDll.SearchEntry(fileData)) {
242 results.insert({"Search", 1}); // Error finding the entry
243 continue;
244 }
245 fileData.EncryptedFilePath->assign(filePath);
246 fileDataVec.push_back(fileData);
247 }
248
249 // Decrypt the files
250 std::vector<bool> decryptResults(fileDataVec.size(), false);
251 cryptoDll.DecryptFiles(fileDataVec.data(), static_cast<int>(fileDataVec.size()), decryptResults);
252
253 for (size_t i = 0; i < fileDataVec.size(); ++i) {
254 std::string algorithm(fileDataVec[i].AlgorithmenType->begin(), fileDataVec[i].AlgorithmenType->end());
255
256 // Check if decryption was successful
257 if (!decryptResults[i]) {
258 results.insert({algorithm, 2}); // Error during decryption
259 continue;
260 }
261
262 // Delete the entry from the database
263 if (!restApiDll.DeleteEntry(fileDataVec[i])) {
264 results.insert({algorithm, 3}); // Error deleting the entry
265 continue;
266 }
267
268 // Clean up the FileData structure
269 fileDataVec[i].Key->clear();
270 fileDataVec[i].Iv->clear();
271 results.insert({algorithm, -1}); // Success
272 }
273
274 return results;

284std::vector<int> HelperUtils::encryptFiles(const std::vector<fs::path> &filePaths, const std::vector<std::string> &algorithm) {
285 CryptoDLL cryptoDll;
286 RestApiDLL restApiDll;
287 AttributeUtils structUtils;
288 std::vector<int> results;
289 std::vector<FileData> fileDataVec;
290
291 for (int i = 0; i < filePaths.size(); i++) {
292 // Create the FileData structure

295 structUtils.createFileDataStruct(globalDefinitions::AlgorithmType::AES128, filePaths[i], fileData);

303 std::vector<bool> encryptResults(fileDataVec.size(), false);
304 cryptoDll.EncryptFiles(fileDataVec.data(), static_cast<int>(fileDataVec.size()), encryptResults);
305
306 for (auto fileData : fileDataVec) {
307 // Check if encryption was successful
308 if (fileData.EncryptedFilePath->empty() || !encryptResults[0]) {
309 results.push_back(1); // Error during encryption
310 continue;
311 }
312
313 // Insert the entry into the database
314 if (!restApiDll.InsertEntry(fileData)) {
315 results.push_back(2); // Error inserting the entry
316 continue;
317 }
318
319 // Clean up the FileData structure
320 fileData.Key->clear();
321 fileData.Iv->clear();
322 results.push_back(-1); // Success
323 }
324
325 return results;
