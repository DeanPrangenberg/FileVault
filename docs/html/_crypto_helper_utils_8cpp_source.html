21void CryptoHelperUtils::deleteFile(const fs::path &filePath) {

27 fs::remove(filePath);

29 std::cout << "Deleting encrypted file after decryption: " << filePath << std::endl;

32 std::cout << "Skipped file deletion for: " << filePath << std::endl;

47bool CryptoHelperUtils::MarkFile(const class FileData *fileData) {

64 unloadDll(hFileMarkDll);
