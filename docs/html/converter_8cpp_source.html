14wchar_t* RestApiDLL::convertToHexWChar(const unsigned char* input, size_t size) {
15 if (input == nullptr || size == 0) {
16 std::cerr << "convertToHexWChar: Input is null or size is zero" << std::endl;
17 return nullptr;
18 }
19
20 // Stream for hex representation
21 std::wstringstream wss;
22 for (size_t i = 0; i < size; ++i) {
23 wss << std::setw(2) << std::setfill(L'0') << std::hex << static_cast<int>(input[i]);
24 }
25
26 // Save hex representation as wstring
27 std::wstring hexStr = wss.str();
28
29 // Allocate memory for wchar_t* (including null termination)
30 wchar_t* result = new wchar_t[hexStr.size() + 1];
31 std::wmemcpy(result, hexStr.c_str(), hexStr.size() + 1); // Include null termination
32
33 return result;

43unsigned char* RestApiDLL::convertFromHexWChar(const wchar_t* input, size_t& size) {
44 if (input == nullptr) {
45 std::cerr << "convertFromHexWChar: Input is null" << std::endl;
46 size = 0;
47 return nullptr;
48 }
49
50 size_t len = std::wcslen(input);
51 if (len % 2 != 0) {
52 std::cerr << "convertFromHexWChar: Invalid hex string length" << std::endl;
53 size = 0;
54 return nullptr;
55 }
56
57 size = len / 2; // Every two hex digits correspond to one byte
58 unsigned char* result = new unsigned char[size];
59
60 for (size_t i = 0; i < size; ++i) {
61 unsigned int byte;
62 std::swscanf(input + 2 * i, L"%2x", &byte);
63 result[i] = static_cast<unsigned char>(byte);
64 }
65
66 return result;
67}
