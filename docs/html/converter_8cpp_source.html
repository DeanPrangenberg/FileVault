15wchar_t *RestApiDLL::convertToHexWChar(const unsigned char *input, size_t size) {
16 if (input == nullptr || size == 0) {
17 std::cerr << "convertToHexWChar: Input is null or size is zero" << std::endl;
18 return nullptr;
19 }
20
21 // Stream for hex representation
22 std::wstringstream wss;
23 for (size_t i = 0; i < size; ++i) {
24 wss << std::setw(2) << std::setfill(L'0') << std::hex << static_cast<int>(input[i]);
25 }
26
27 // Save hex representation as wstring
28 std::wstring hexStr = wss.str();
29
30 // Allocate memory for wchar_t* (including null termination)
31 wchar_t *result = new wchar_t[hexStr.size() + 1];
32 std::wmemcpy(result, hexStr.c_str(), hexStr.size() + 1); // Include null termination
33
34 return result;
35}

44std::shared_ptr<std::vector<unsigned char>> RestApiDLL::convertFromHexWChar(const wchar_t *input) {
45 if (input == nullptr) {
46 std::cerr << "convertFromHexWChar: Input is null" << std::endl;
47 return nullptr;
48 }
49
50 size_t len = std::wcslen(input);
51 if (len % 2 != 0) {
52 std::cerr << "convertFromHexWChar: Invalid hex string length" << std::endl;
53 return nullptr;
54 }
55
56 auto result = std::make_shared<std::vector<unsigned char>>(len / 2); // Every two hex digits correspond to one byte
57
58 for (size_t i = 0; i < result->size(); ++i) {
59 unsigned int byte;
60 std::swscanf(input + 2 * i, L"%2x", &byte);
61 (*result)[i] = static_cast<unsigned char>(byte);
62 }
63
64 return result;
65}

176FileData RestApiDLL::convertDBStructToFileData(FileDataDB &data) {
177 FileData fileData; // Annahme: Konstruktor initialisiert alle shared_ptr mit leeren Objekten
178
179 auto validateHexConversion = [](const std::shared_ptr<std::vector<unsigned char>>& result, size_t expectedLength) {
180 if (!result) throw std::runtime_error("Hex-Konvertierung fehlgeschlagen");
181 if (expectedLength > 0 && result->size() != expectedLength) {
182 throw std::runtime_error("Längenmismatch: Erwartet " +
183 std::to_string(expectedLength) +
184 ", erhalten " +
185 std::to_string(result->size()));
186 }
187 };
188
189 try {
190 // AlgorithmenType
191 if (data.AlgorithmenType && data.AlgorithmenType[0] != L'\0') {
192 *fileData.AlgorithmenType = wcharToString(data.AlgorithmenType);
193
194 // Algorithmus-spezifische Validierungen
195 if (*fileData.AlgorithmenType == "AES128" || *fileData.AlgorithmenType == "AES256") {
196 size_t expectedKeyLen = (*fileData.AlgorithmenType == "AES128") ? 16 : 32;
197 if (fileData.Key->size() != expectedKeyLen) {
198 fileData.Key->resize(expectedKeyLen);
199 }
200 if (fileData.Iv->size() != 16) {
201 fileData.Iv->resize(16);
202 }
203 }
204 }
205
206 // FileID
207 if (data.FileID) {
208 fileData.FileID = convertFromHexWChar(data.FileID);
209 validateHexConversion(fileData.FileID, data.FileIDLength);
210 } else {
211 Logs::writeToErrorLog("Conversion from DB to FileData: FileID is null");
212 }
213
214 // EncryptedID
215 if (data.EncryptedID) {
216 fileData.EncryptionID = convertFromHexWChar(data.EncryptedID);
217 validateHexConversion(fileData.EncryptionID, data.EncryptedIDLength);
218 } else {
219 Logs::writeToErrorLog("Conversion from DB to FileData: EncryptedID is null");
220 }
221
222 // LastUpdateID
223 if (data.LastUpdateID) {
224 fileData.LastUpdateID = convertFromHexWChar(data.LastUpdateID);
225 validateHexConversion(fileData.LastUpdateID, data.LastUpdateIDLength);
226 }
227
228 // Key & IV mit doppelter Validierung
229 if (data.Key) {
230 fileData.Key = convertFromHexWChar(data.Key);
231 validateHexConversion(fileData.Key, data.KeyLength);
232
233 // Zusätzliche Algorithmus-Validierung
234 if (!fileData.AlgorithmenType->empty()) {
235 if (*fileData.AlgorithmenType == "AES128" && fileData.Key->size() != 16) {
236 throw std::runtime_error("AES128 benötigt 16-Byte-Key");
237 }
238 if (*fileData.AlgorithmenType == "AES256" && fileData.Key->size() != 32) {
239 throw std::runtime_error("AES256 benötigt 32-Byte-Key");
240 }
241 }
242 }
243
244 if (data.Iv) {
245 fileData.Iv = convertFromHexWChar(data.Iv);
246 validateHexConversion(fileData.Iv, data.IvLength);
247 if (fileData.Iv->size() != 16) {
248 throw std::runtime_error("IV muss 16 Byte lang sein");
249 }
250 }
251
252 // Pfad-Konvertierung mit Nullptr-Check
253 auto convertPath = [](const wchar_t* path) {
254 return path ? std::make_shared<fs::path>(path) : std::make_shared<fs::path>();
255 };
256
257 fileData.OriginalFilePath = convertPath(data.OriginalFilePath);
258 fileData.EncryptedFilePath = convertPath(data.EncryptedFilePath);
259 fileData.DecryptedFilePath = convertPath(data.DecryptedFilePath);
260
261 } catch (const std::exception& e) {
262 std::cerr << "Konvertierungsfehler: " << e.what() << std::endl;
263 }
264
265 return fileData;
266}
