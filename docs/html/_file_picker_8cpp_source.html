248void FilePicker::scanDirectoryInThread(const QString &directory) {
249
250 // Start a new thread for scanning the directory
251 std::thread([this, directory]() {
252 fs::path dirPath = directory.toStdWString();
253 std::vector<fs::path> scannedPaths;
254
255 fileScanner.ScanDirectory(dirPath, filePickerType == FilePickerType::DECRYPT, scannedPaths, [this](const fs::path &path) {
256 QString qPath = QString::fromStdWString(path.wstring());
257 if (!addedPaths.contains(qPath)) {
258 addedPaths.insert(qPath);
259 QMetaObject::invokeMethod(this, [this, qPath]() {
260 addEntry(qPath);
261 }, Qt::QueuedConnection);
262 }
263 });
264 }).detach(); // Detach the thread to run independently
265}
