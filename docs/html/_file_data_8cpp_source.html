99std::string FileData::toHexString(const unsigned char *pUChar, size_t length) {
100 if (pUChar == nullptr) {
101 throw std::invalid_argument("Null pointer passed to toHexString");
102 }
103
104 // Optimierte String-Speicherreservierung
105 std::ostringstream oss;
106 oss << std::hex << std::setw(2) << std::setfill('0');
107
108 for (size_t i = 0; i < length; ++i) {
